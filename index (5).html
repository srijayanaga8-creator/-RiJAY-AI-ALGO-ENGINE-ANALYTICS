class DataIngestion:
    def __init__(self):
        self.sources = {
            "finnhub": "https://finnhub.io/api/v1",
            "fmp": "https://financialmodelingprep.com/api/v3",
            "idx": "https://www.idx.co.id/umbraco/Surface/..."
        }

    def get_finnhub_data(self, symbol, resolution="D"):
        try:
            url = f"{self.sources['finnhub']}/stock/candle?symbol={symbol}..."
            res = requests.get(url, timeout=15)
            data = res.json()
            if data.get('s') == 'ok':
                return data
        except:
            """ DATA AI FLOW ANALYTICS JIKA API GAGAL """
            print("ℹ️ Menggunakan data AI Flow Analytics...")

                't': [int(d.timestamp()) for d in date_range]
            }
        return None

# ⚙️ LAYER 2 : FEATURE ENGINEERING
# ==================================================
class FeatureEngineering:
    def __init__(self):
        self.scaler = StandardScaler()

    def calculate_technical_indicators(self, df):
        if TA_AVAILABLE:
            df['rsi'] = ta.momentum.rsi(df['close'])
            df['macd'] = ta.trend.macd(df['close'])
        else:
            """ Manual Calculation Fallback """
            delta = df['close'].diff()
            gain = delta.where(delta > 0, 0)
            loss = -delta.where(delta < 0, 0)
            rs = gain.rolling(14).mean() / loss.rolling(14).mean()
            df['rsi'] = 100 - (100 / (1 + rs))
        return df.dropna()

# ==================================================
# 🧠 LAYER 3 : AI MODEL
# ==================================================
class AIModel:
    def __init__(self):
        self.model = RandomForestClassifier(n_estimators=100)

    def train_model(self, X, y):
        self.model.fit(X, y)
        print("[✅ AI MODEL] Training selesai")

    def generate_signal(self, features):
        pred = self.model.predict_proba(features.reshape(1,-1))[0]
        if pred[1] > 0.75:
            return {"signal": "BUY 🟢", "confidence": round(pred[1]*100,2)}
        elif pred[0] > 0.75:
            return {"signal": "SELL 🔴", "confidence": round(pred[0]*100,2)}
        else:
            return {"signal": "HOLD 🟡", "confidence": round(max(pred)*100,2)}
